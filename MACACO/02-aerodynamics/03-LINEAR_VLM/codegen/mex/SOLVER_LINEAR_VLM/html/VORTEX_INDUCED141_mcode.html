1function [Vind] = VORTEX_INDUCED(P,Vortex,G,tipo)
  2%{
  3
  4Função responsável por calcular a velocidade em um ponto P devido a um
  5Vortice de intensidade G.
  6
  7Vortex(	1:k				,	1:3		)
  8		k-ésimo vértice		coordenadas
  9		do vortice			x,y,z
 10
 11
 12
 13Exemplo de vórtice linear (tipo = 0)
 14
 15-OO
 16
 17k=2	o
 18	^
 19	^
 20	^
 21	^	P	
 22	^	*	
 23k=1	o	
 24	
 25+OO         
 26
 27
 28Exemplo de vórtice ferradura (tipo = 1)
 29
 30k=3	o > > > o k=4
 31	^		v
 32	^		v
 33	^		v
 34	^	P	v
 35	^	*	v
 36k=2	o		o k=5
 37    ^		v
 38    ^		v
 39    ^		v
 40    ^		v
 41k=1	o       o k=6
 42
 43+OO          +OO
 44
 45
 46
 47Exemplo de vórtice anel (tipo = 2)
 48
 49k=2	o > > > o k=3
 50	^		v
 51	^		v
 52	^	P	v
 53	^	*	v
 54	^		v
 55k=1	o < < < o k=4
 56        
 57%}
 58
 59K = length(Vortex(1,:,1));
 60Vind_k = zeros(K,3);
 61switch tipo
 62	
 63	case 1
 64		V1 = zeros(1,3);
 65		V2 = zeros(1,3);
 66		
 67% 		Vind_k = zeros(K,3);
 68		
 69		% Cálculo da velocidade induzida por cada segmento do vórtice
 70		for k=1:K-1
 71			
 72			V1(:) = Vortex(1,k  ,:);	
 73			V2(:) = Vortex(1,k+1,:);
 74
 75			r0 = V2-V1;
 76			r1 = P-V1;
 77			r2 = P-V2;
 78						
 79			r0_s = norm(r0);
 80			r1_s = norm(r1);
 81			r2_s = norm(r2);
 82			
 83			% Diâmetro do núcleo do vórtice
 84			D = 1e-9;
 85			if any(abs([r0_s, r1_s, r2_s])<=D)
 86				Vind_k(k,:) = 0;
 87			else	
 88				r1xr2   = cross(r1,r2);
 89				r1xr2_s = norm(r1xr2);
 90
 91				Vind_k(k,:) = G/(4*pi) * (r1xr2)/(r1xr2_s^2) *  dot(r0, (r1/r1_s - r2/r2_s));
 92			end
 93		end
 94		
 95		
 96		
 97end
 98
 99% Soma das velocidades induzidas por cada segmento
100Vind = sum(Vind_k);
